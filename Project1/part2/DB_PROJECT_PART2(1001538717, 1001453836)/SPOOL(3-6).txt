--3.11) Write a query that combines the results of the queries in 9. and 10. 
--To get the total number of games each country has played (either as Team1 or as Team2), 
--their total goals scored and their total goals against. 
--Create a view TEAM_SUMMARY that has the following data attributes to hold the 
--result of the combined query: CountryName, NoOfGames, TotalGoalsFor, TotalGoalsAgainst. Order in descending order of number of games played.
Create view TEAM_SUMMARY(Country_Name,num_of_games_played,total_goals_scored,goals_against)
  2  AS
  3  SELECT
  4  Country_Name,
  5  sum(num_of_games_played),
  6  sum(total_goals_scored),
  7  sum(goals_against)
  8  FROM
  9  ((select COUNTRY.Country_Name,count(*) as num_of_games_played, sum(MATCH_RESULTS.Team1_score) as total_goals_scored ,sum(MATCH_RESULTS.Team2_score) as goals_against
 10  FROM COUNTRY,MATCH_RESULTS WHERE COUNTRY.Country_Name=MATCH_RESULTS.Team1
 11  GROUP BY COUNTRY.Country_Name
 12  )
 13   UNION
 14   (select COUNTRY.Country_Name,count(*) as num_of_games_played, sum(MATCH_RESULTS.Team2_score) as total_goals_scored ,
 15   sum(MATCH_RESULTS.Team1_score) as goals_against FROM COUNTRY,MATCH_RESULTS
 16   WHERE COUNTRY.Country_Name=MATCH_RESULTS.Team2 GROUP BY COUNTRY.Country_Name
 17   ))
 18   GROUP BY Country_Name;
